│  │  │    └────────────────────────────────────────────┘          │   │   │

│  │  │  private fun parseMessage(text: String): StockPrice? {     │   │   │
│  │  │      val json = JSONObject(text)                           │   │   │
│  │  │      if (json.getString("e") == "aggTrade") {              │   │   │
│  │  │        StockPrice(                                          │   │   │
│  │  │          ticker = json.getString("s"),                     │   │   │
│  │  │          price = json.getDouble("p"),                      │   │   │
│  │  │          timestamp = format(json.getLong("T"))             │   │   │
│  │  │        )                                                    │   │   │
│  │  │      }                                                      │   │   │
│  │  │    }                                                        │   │   │
│  │  │                                                             │   │   │
│  │  │  override fun disconnect() {                              │   │   │
│  │  │      webSocket?.close(1000, "Disconnected...")            │   │   │
│  │  │      webSocket = null                                      │   │   │
│  │  │  }                                                          │   │   │
│  │  └─────────────────────────────────────────────────────────────┘   │   │
│  │                              │                                        │   │

│  │  │                                                             │   │   │
│  │  │  override fun onOpen(ws: WebSocket, response: Response) { │   │   │

│  │  │      put("id", 1)                                          │   │   │
│  │  │    }                                                        │   │   │

│                          DATA FLOW TIMELINE                              │
├──────────────────────────────────────────────────────────────────────────┤
│                                                                          │
│  1. INITIALIZATION                                                      │
│     ┌──────────────┐                                                   │
│     │ MainActivity │ onCreate()                                        │
│     └──────┬───────┘                                                   │
│            │                                                            │
│            ▼                                                            │
│     ┌─────────────────┐                                                │
│     │ LaunchedEffect  │                                                │
│     │ observeStocks() │──────────────────┐                              │
│     └─────────────────┘                  │                              │
│                                          ▼                              │
│     ┌─────────────────────────────────────────────────────┐            │
│     │ StockViewModel.observeStocks()                      │            │
│     │  • repository.getStockPriceStream(symbols)          │            │
│     │    └─> callbackFlow in BinanceStockRepositoryImpl  │            │
│     │  • shareIn(WhileSubscribed(5000))                  │            │
│     │  • flatMapLatest { if (throttled) sample(500) }    │            │
│     │  • collect { update uiState }                      │            │
│     └─────────────────────────────────────────────────────┘            │
│                                          │                              │
│  2. WEBSOCKET CONNECTION                                             │
│                                          ▼                              │
│     ┌─────────────────────────────────────────────────────┐            │
│     │ BinanceStockRepositoryImpl.getStockPriceStream()   │            │
│     │  • Creates WebSocket                               │            │
│     │    Request("wss://stream.binance.com:9443/ws")    │            │
│     │    with BinanceWebSocketListener                  │            │
│     └──────────────────────────┬────────────────────────┘            │
│                                  │                                    │
│                                  ▼                                    │
│     ┌──────────────────────────────────────────────┐                 │
│     │ BinanceWebSocketListener.onOpen()           │                 │
│     │ • Send SUBSCRIBE message                    │                 │
│     │   {"method":"SUBSCRIBE","params":           │                 │
│     │    ["btcusdt@aggTrade","ethusdt@aggTrade", │                 │
│     │     "bnbusdt@aggTrade","solusdt@aggTrade", │                 │
│     │     "adausdt@aggTrade"],"id":1}            │                 │
│     └──────────────────────────────────────────────┘                 │
│                                                                          │
│  3. INCOMING MESSAGE (Per Trade Event)                                │
│                                          │                              │
│     ┌─────────────────────────────────────────────────────┐            │
│     │  Binance WebSocket sends aggTrade event:          │            │
│     │  { "e":"aggTrade", "s":"BTCUSDT",                │            │
│     │    "p":"67245.12", "T":1710987654321 }           │            │
│     └───────────────────────────────────────────┬─────┘            │
│                                                  │                   │
│     ┌───────────────────────────────────────────▼─────┐            │
│     │ BinanceWebSocketListener.onMessage(text)       │            │
│     │ → onMessageReceived(text) callback             │            │
│     └───────────────────────────────────────────┬─────┘            │
│                                                  │                   │
│     ┌───────────────────────────────────────────▼─────┐            │
│     │ BinanceStockRepositoryImpl.parseMessage(text)  │            │
│     │ • JSON extraction                               │            │
│     │ • Create StockPrice(                            │            │
│     │     ticker="BTCUSDT",                           │            │
│     │     price=67245.12,                             │            │
│     │     timestamp="HH:mm:ss" )                      │            │
│     └───────────────────────────────────────────┬─────┘            │
│                                                  │                   │
│     ┌───────────────────────────────────────────▼─────┐            │
│     │ callbackFlow.trySend(StockPrice)               │            │
│     │ → Flow emits StockPrice object                 │            │
│     └───────────────────────────────────────────┬─────┘            │
│                                                  │                   │
│  4. FLOW PROCESSING IN VIEWMODEL                                      │
│                                                  │                   │
│     ┌───────────────────────────────────────────▼─────┐            │
│     │ StockViewModel.collect from repository flow    │            │
│     │                                                  │            │
│     │  flatMapLatest on isThrottlingEnabled:         │            │
│     │    • true  → sharedFlow.sample(500)           │            │
│     │    • false → sharedFlow directly              │            │
│     │                                                  │            │
│     │  For each StockPrice received:                 │            │
│     │    • Get current history for ticker            │            │
│     │    • Append new price: history + price         │            │
│     │    • Limit to last 50: takeLast(50)           │            │
│     │    • Update uiState.stockPrices map            │            │
│     │    • isLoading = false, error = null           │            │
│     └───────────────────────────────────────────┬─────┘            │
│                                                  │                   │
│  5. UI RENDERING                                                      │
│                                                  │                   │
│     ┌───────────────────────────────────────────▼─────┐            │
│     │ MainActivity.collectAsStateWithLifecycle(uiState)│           │
│     │  • Re-composes when uiState changes              │           │
│     └───────────────────────────────────────────┬─────┘            │
│                                                  │                   │
│     ┌───────────────────────────────────────────▼─────┐            │
│     │ StockList Component                            │            │
│     │  • For each (ticker, history) in stockPrices:  │            │
│     │    - Get last price: history.last()            │            │
│     │    - Render StockListItem(ticker, price)       │            │
│     └───────────────────────────────────────────┬─────┘            │
│                                                  │                   │
│     │     On item click                            │                   │
│     │     navController.navigate("stock_details/$ticker")            │
│     │                                                  │            │
│     └───────────────────────────────────────────┬─────┘            │
│                                                  │                   │
│     ┌───────────────────────────────────────────▼─────┐            │
│     │ StockDetailsScreen(ticker)                       │            │
│     │  • Get history = uiState.stockPrices[ticker]    │            │
│     │  • currentPrice = history.lastOrNull()          │            │
│     │  • Show: Price, Throttle status, Statistics     │            │
│     │  • Render PriceGraph(prices = history)          │            │
│     └───────────────────────────────────────────┬─────┘            │
│                                                  │                   │
│     ┌───────────────────────────────────────────▼─────┐            │
│     │ PriceGraph (Canvas)                             │            │

## Memory Footprint & Performance

┌────────────────────────────────────────────────────────────────────────────┐
│                      PERFORMANCE CHARACTERISTICS                         │
├────────────────────────────────────────────────────────────────────────────┤
│                                                                             │
│  WebSocket Connection:                                                    │
│    • Single shared connection via shareIn()                              │
│    • WhileSubscribed(5000) keeps alive 5s after last observer          │
│    • Auto-reconnects on failure                                         │
│                                                                             │
│  Data Flow:                                                               │
│    • Real-time: Every aggTrade event → UI instantly                    │
│    • Throttled: sample(500) = max 2 updates/sec per ticker             │
│    • Buffer: Maintains last 50 prices per ticker in memory             │
│      (5 tickers × 50 × ~40 bytes = ~10KB)                              │
│                                                                             │
│  UI Rendering:                                                           │
│    • Compose recomposes on StateFlow changes (per ticker updates)      │
│    • PriceGraph: O(n) path construction, n = history length (≤50)     │
│    • LazyColumn: recycles item composables                             │
│                                                                             │
│  Lifecycle:                                                              │
│    • ViewModel survives config changes                                 │
│    • WebSocket closed when ViewModel.onCleared()                       │
│    • Flow collection auto-stops with collectAsStateWithLifecycle()    │
│                                                                             │
└────────────────────────────────────────────────────────────────────────────┘
